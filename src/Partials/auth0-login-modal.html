       var failedAttempts = 0;

       lock.show({
           authParams: {
               scope: 'openid profile'
           },
           container: 'lock-widget',
           callbackURL: window.location.origin + '/signin-auth0',
           disableSignupAction: true,
           responseType: 'code',
           closable: true
       });

       lock.on('signin error', function () {
           failedAttempts++;

           if (failedAttempts > 5) {
               lock.$dicts.en.signin.wrongEmailPasswordErrorText = 'Wrong email or password.<br /><br />Warning! Your account will be locked after 10 failed attempts!';
           }
       });
   </script>
